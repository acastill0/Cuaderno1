En base a la simulacion, generar una simulacion lo mas apegado a la realidad del Covid-19, para ello obtener informacion del numero de consultorios y camas disponibles en un hospital del Ecuador. Parametrizar la simulacion para ingresar el numero de pacientes, el numero de consultorios y el numero de camas y algun otro parametro que considere oportuno.

Finalmente generar las siguientes metricas :

  • Tiempo de espera promedio para cada sala.
  • Tiempo promedio total que requiere un paciente para ser atendido desde cero.
  • Cuntos pacientes se atendieron.
  • Cuntos pacientes se quedaron sin atender y fallecieron.

persona_recuperadas={}

class Hospital(object):

    def __init__(self, env, num_cama,num_consultorios, name):
            self.env = env
            self.num_cama = num_cama
            self.num_consultorios=num_consultorios
            self.camas = simpy.Resource(env, num_cama)
            self.consultorios=simpy.Resource(env, num_consultorios)
            self.name = name
            
    def consulta(self,paciente):
        duracion=random.uniform(TIEMPO_CONSULTA-0.005, TIEMPO_CONSULTA+0.005)
        yield self.env.timeout(duracion)
        print('\nPaciente N', paciente,'ingreso al cosultorio en tiempo de: ', int(duracion*1440),'min')
        timeConsultas.append(duracion)
        
    def ingresar_paciente(self, paciente):
        yield self.env.timeout(random.randint(DIAS_INTERNADO-5, DIAS_INTERNADO+5))
        print("\nPaciente N", paciente, " se interno el da ", int(self.env.now))

def llegada_paciente(env, hospital, paciente):
    arrive = env.now
    with hospital.consultorios.request() as consultorio:
        espera_consulta=random.randint(1,2)
        timeEspera.append(espera_consulta)
        demanda = yield consultorio | env.timeout(espera_consulta)
        llegadaC = env.now - arrive
        if consultorio in demanda:
            print("Paciente N", paciente, " se le asigna una cama ")
            yield env.process(hospital.consulta(paciente))
            estado = random.randint(1,100)
            atendidos.append(1)
            if(estado < 40):
                with hospital.camas.request() as cama:
                    llegadaC = env.now
                    dias_esperando = random.randint(1,5) #Numero de dias maximo que puede esperar sin ser asignado a una cama dentro del hospital
                    requerimiento = yield cama | env.timeout(dias_esperando) #Asignar un tiempo de espera
                    wait = env.now - llegadaC
                    if cama in requerimiento:
                        print("Paciente N", paciente, " se le asigna una cama ")
                        yield env.process(hospital.ingresar_paciente(paciente))
                        estado = random.randint(1,100)
                        if (estado < 8) :
                            persona_fallecidas[env.now] = persona_fallecidas[env.now] + 1 if env.now in persona_fallecidas else  1
                        else:
                            persona_recuperadas[env.now] = persona_recuperadas[env.now] + 1 if env.now in persona_recuperadas else 1
                    else:
                        print("Paciente N" , paciente," espero ", wait , " das y fallece por falta de camas")
                        persona_fallecidas[env.now] = persona_fallecidas[env.now] + 1 if env.now in persona_fallecidas else  1
            else :
                print('Paciente N',paciente,'no tiene COVID-19')
        else:
            print('\nPaciente N' , paciente,' espero ', llegadaC , " das y fallece por no ingresar a consulta")
            noAtendidosMuertos.append(1)
            persona_fallecidas[env.now] = persona_fallecidas[env.now] + 1 if env.now in persona_fallecidas else  1

def ejecutar(env, tasa_crecimiento, infectados):
    hospital = Hospital(env, CAMAS,CONSULTORIOS, "")
    for i in range(infectados):
        asignar_hospital(env, hospital, i)
    paciente = infectados
    while True:
        yield env.timeout(1)
        for i in range(tasa_crecimiento):
            paciente += 1
            asignar_hospital(env, hospital, paciente)
                
def asignar_hospital(env, hospital, paciente):
    hosp_esc = 1
    if (hosp_esc == 1):
        print("Llega paciente N: ", paciente, " en tiempo ", env.now)
        env.process(llegada_paciente(env, hospital, paciente))
